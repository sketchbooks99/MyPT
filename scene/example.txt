# Example scene (simple ground plane with checker texture and glass sphere)

filename result/example.png
# Image resolution
width 640   
height 320
# The number of samples per pixel
spp 1
# The number of maximum depth to track rays
depth 5
# Background color
background 0.7 0.8 0.9

# Camera settings
beginCamera
origin 10 10 -70
lookat 0 0 0 
up 0 1 0
focus_length 15.0
aperture 0
endCamera

# Ground plane
beginPrimitive 
shape plane min -100 -100 max 100 100
translate 0 -10 0
material lambertian checker color1 0.3 0.3 0.3 color2 0.9 0.9 0.9 scale 100
endPrimitive

# Sphere
beginPrimitive
shape sphere radius 5
translate -1 -3 0
material dielectric color 1 1 1 ior 1.5
endPrimitive

beginPrimitive 
shape sphere radius 5
translate -12 -3 0
material metal color 0.8 0.05 0.05 fuzz 0.03 
endPrimitive

beginPrimitive 
shape sphere radius 5 
translate 10 -3 0
material lambertian color 0.05 0.8 0.05
endPrimitive

beginPrimitive 
shape mesh filename data/model/bunny.obj axis 1 1 1 smooth
translate 0 -5 10 
rotate_y 180
scale 75
material lambertian color 0.8 0.8 0.8
endPrimitive
